SET FOREIGN_KEY_CHECKS = 0;

DROP TABLE IF EXISTS jegyek;
DROP TABLE IF EXISTS Tavollet;
DROP TABLE IF EXISTS Feladat;
DROP TABLE IF EXISTS projekt;
DROP TABLE IF EXISTS Alkalmazott;
DROP TABLE IF EXISTS Ugyfel;
DROP TABLE IF EXISTS Csapat;
DROP TABLE IF EXISTS Felhasznalo;

SET FOREIGN_KEY_CHECKS = 1;

CREATE TABLE Felhasznalo (
    Id BIGINT NOT NULL AUTO_INCREMENT,
    Keresztnev VARCHAR(255),
    Vezeteknev VARCHAR(255),
    Email VARCHAR(255) NOT NULL,
    Jelszo VARCHAR(255) NOT NULL,
    Szuletesi_datum DATE,
    Telefonszam VARCHAR(255),
    Cim VARCHAR(255),
    Varos VARCHAR(255),
    Iranyitoszam VARCHAR(255),
    Hazszam VARCHAR(255),
    Szerepkor VARCHAR(50),
    Aktiv BIT NOT NULL DEFAULT 0,
    Visszaallito_kod VARCHAR(255),
    Visszaallito_kod_lejar DATETIME(6),
    Aktivalo_kod VARCHAR(255),
    Aktivalo_kod_lejar DATETIME(6),
    PRIMARY KEY (Id),
    UNIQUE KEY uk_felhasznalo_email (Email)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

CREATE TABLE Alkalmazott (
    Id BIGINT NOT NULL,
    Adoazonosito VARCHAR(255),
    Szemelyi_igazolvany_szam VARCHAR(255),
    Taj_szam VARCHAR(255),
    Fizetes BIGINT,
    Felvetel_datum DATE,
    Csapat_id BIGINT,
    Teljesitett_feladatok INT NOT NULL DEFAULT 0,
    Lezart_jegyek INT NOT NULL DEFAULT 0,
    PRIMARY KEY (Id),
    CONSTRAINT fk_alkalmazott_felhasznalo
        FOREIGN KEY (Id) REFERENCES Felhasznalo (Id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

CREATE TABLE Ugyfel (
    Id BIGINT NOT NULL,
    Ceg_nev VARCHAR(255) NOT NULL,
    PRIMARY KEY (Id),
    CONSTRAINT fk_ugyfel_felhasznalo
        FOREIGN KEY (Id) REFERENCES Felhasznalo (Id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

CREATE TABLE Csapat (
    Id BIGINT NOT NULL AUTO_INCREMENT,
    Nev VARCHAR(255),
    Tulajdonos_id BIGINT NOT NULL,
    PRIMARY KEY (Id),
    UNIQUE KEY uk_csapat_tulajdonos (Tulajdonos_id),
    CONSTRAINT fk_csapat_tulajdonos
        FOREIGN KEY (Tulajdonos_id) REFERENCES Alkalmazott (Id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

ALTER TABLE Alkalmazott
    ADD CONSTRAINT fk_alkalmazott_csapat
        FOREIGN KEY (Csapat_id) REFERENCES Csapat (Id);

CREATE TABLE projekt (
    Id BIGINT NOT NULL AUTO_INCREMENT,
    Letrehozo_id BIGINT NOT NULL,
    Termek_tulajdonos_id BIGINT,
    Csapat_id BIGINT,
    Cim VARCHAR(255),
    Leiras VARCHAR(255),
    Statusz VARCHAR(50),
    Projekt_koltseg BIGINT NOT NULL DEFAULT 0,
    Kezdete DATE,
    Hatarido DATE,
    Befejezve DATE,
    PRIMARY KEY (Id),
    KEY idx_projekt_statusz (Statusz),
    KEY idx_projekt_befejezve (Befejezve),
    CONSTRAINT fk_projekt_letrehozo
        FOREIGN KEY (Letrehozo_id) REFERENCES Felhasznalo (Id),
    CONSTRAINT fk_projekt_termek_tulajdonos
        FOREIGN KEY (Termek_tulajdonos_id) REFERENCES Alkalmazott (Id),
    CONSTRAINT fk_projekt_csapat
        FOREIGN KEY (Csapat_id) REFERENCES Csapat (Id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

CREATE TABLE Feladat (
    Id BIGINT NOT NULL AUTO_INCREMENT,
    Projekt_id BIGINT NOT NULL,
    Munkatars_id BIGINT,
    Cim VARCHAR(255),
    Leiras VARCHAR(255),
    Statusz VARCHAR(50),
    Prioritas VARCHAR(50),
    PRIMARY KEY (Id),
    KEY idx_feladat_statusz (Statusz),
    CONSTRAINT fk_feladat_projekt
        FOREIGN KEY (Projekt_id) REFERENCES projekt (Id),
    CONSTRAINT fk_feladat_munkatars
        FOREIGN KEY (Munkatars_id) REFERENCES Alkalmazott (Id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

CREATE TABLE Tavollet (
    Id BIGINT NOT NULL AUTO_INCREMENT,
    Munkatars_id BIGINT,
    Jovahagyta BIGINT,
    Tipus VARCHAR(50),
    Statusz VARCHAR(50),
    Kezdet DATE,
    Vege DATE,
    Munkanapok INT NOT NULL DEFAULT 0,
    PRIMARY KEY (Id),
    KEY idx_tavollet_statusz (Statusz),
    KEY idx_tavollet_kezdet (Kezdet),
    CONSTRAINT fk_tavollet_munkatars
        FOREIGN KEY (Munkatars_id) REFERENCES Alkalmazott (Id),
    CONSTRAINT fk_tavollet_jovahagyta
        FOREIGN KEY (Jovahagyta) REFERENCES Alkalmazott (Id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

CREATE TABLE jegyek (
    Id BIGINT NOT NULL AUTO_INCREMENT,
    Letrehozta BIGINT NOT NULL,
    It_munkatars_id BIGINT,
    Tipus VARCHAR(50),
    Statusz VARCHAR(50),
    Problema TEXT,
    PRIMARY KEY (Id),
    KEY idx_jegyek_statusz (Statusz),
    CONSTRAINT fk_jegyek_letrehozta
        FOREIGN KEY (Letrehozta) REFERENCES Felhasznalo (Id),
    CONSTRAINT fk_jegyek_it_munkatars
        FOREIGN KEY (It_munkatars_id) REFERENCES Alkalmazott (Id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;
