SET FOREIGN_KEY_CHECKS = 0;

DROP TABLE IF EXISTS tickets;
DROP TABLE IF EXISTS day_offs;
DROP TABLE IF EXISTS tasks;
DROP TABLE IF EXISTS projects;
DROP TABLE IF EXISTS employees;
DROP TABLE IF EXISTS customers;
DROP TABLE IF EXISTS teams;
DROP TABLE IF EXISTS users;

SET FOREIGN_KEY_CHECKS = 1;

CREATE TABLE users (
    id BIGINT NOT NULL AUTO_INCREMENT,
    first_name VARCHAR(255),
    last_name VARCHAR(255),
    email VARCHAR(255) NOT NULL,
    password VARCHAR(255) NOT NULL,
    birth_date DATE,
    phone_number VARCHAR(255),
    address VARCHAR(255),
    city VARCHAR(255),
    postal_code VARCHAR(255),
    house_number VARCHAR(255),
    role VARCHAR(50),
    active BIT NOT NULL DEFAULT 0,
    reset_code VARCHAR(255),
    reset_code_expires_at DATETIME(6),
    activation_code VARCHAR(255),
    activation_code_expires_at DATETIME(6),
    PRIMARY KEY (id),
    UNIQUE KEY uk_users_email (email)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

CREATE TABLE employees (
    id BIGINT NOT NULL,
    tax_id VARCHAR(255),
    identity_card_number VARCHAR(255),
    social_security_number VARCHAR(255),
    salary BIGINT,
    hire_date DATE,
    team_id BIGINT,
    completed_tasks INT NOT NULL DEFAULT 0,
    closed_tickets INT NOT NULL DEFAULT 0,
    PRIMARY KEY (id),
    CONSTRAINT fk_employees_user
        FOREIGN KEY (id) REFERENCES users (id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

CREATE TABLE customers (
    id BIGINT NOT NULL,
    company_name VARCHAR(255),
    PRIMARY KEY (id),
    CONSTRAINT fk_customers_user
        FOREIGN KEY (id) REFERENCES users (id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

CREATE TABLE teams (
    id BIGINT NOT NULL AUTO_INCREMENT,
    name VARCHAR(255),
    owner_id BIGINT NOT NULL,
    PRIMARY KEY (id),
    UNIQUE KEY uk_teams_owner (owner_id),
    CONSTRAINT fk_teams_owner
        FOREIGN KEY (owner_id) REFERENCES employees (id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

ALTER TABLE employees
    ADD CONSTRAINT fk_employees_team
        FOREIGN KEY (team_id) REFERENCES teams (id);

CREATE TABLE projects (
    id BIGINT NOT NULL AUTO_INCREMENT,
    creator_id BIGINT NOT NULL,
    product_owner_id BIGINT,
    team_id BIGINT,
    title VARCHAR(255),
    description VARCHAR(255),
    status VARCHAR(50),
    budget BIGINT NOT NULL DEFAULT 0,
    start_date DATE,
    deadline DATE,
    completed_at DATE,
    PRIMARY KEY (id),
    KEY idx_projects_status (status),
    KEY idx_projects_completed_at (completed_at),
    CONSTRAINT fk_projects_creator
        FOREIGN KEY (creator_id) REFERENCES users (id),
    CONSTRAINT fk_projects_product_owner
        FOREIGN KEY (product_owner_id) REFERENCES employees (id),
    CONSTRAINT fk_projects_team
        FOREIGN KEY (team_id) REFERENCES teams (id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

CREATE TABLE tasks (
    id BIGINT NOT NULL AUTO_INCREMENT,
    project_id BIGINT NOT NULL,
    employee_id BIGINT,
    title VARCHAR(255),
    description VARCHAR(255),
    status VARCHAR(50),
    priority VARCHAR(50),
    PRIMARY KEY (id),
    KEY idx_tasks_status (status),
    CONSTRAINT fk_tasks_project
        FOREIGN KEY (project_id) REFERENCES projects (id),
    CONSTRAINT fk_tasks_employee
        FOREIGN KEY (employee_id) REFERENCES employees (id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

CREATE TABLE day_offs (
    id BIGINT NOT NULL AUTO_INCREMENT,
    employee_id BIGINT,
    approved_by_id BIGINT,
    type VARCHAR(50),
    status VARCHAR(50),
    start_date DATE,
    end_date DATE,
    workdays INT NOT NULL DEFAULT 0,
    PRIMARY KEY (id),
    KEY idx_day_offs_status (status),
    KEY idx_day_offs_start_date (start_date),
    CONSTRAINT fk_day_offs_employee
        FOREIGN KEY (employee_id) REFERENCES employees (id),
    CONSTRAINT fk_day_offs_approved_by
        FOREIGN KEY (approved_by_id) REFERENCES employees (id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

CREATE TABLE tickets (
    id BIGINT NOT NULL AUTO_INCREMENT,
    created_by_id BIGINT NOT NULL,
    it_employee_id BIGINT,
    type VARCHAR(50),
    status VARCHAR(50),
    problem TEXT,
    PRIMARY KEY (id),
    KEY idx_tickets_status (status),
    CONSTRAINT fk_tickets_created_by
        FOREIGN KEY (created_by_id) REFERENCES users (id),
    CONSTRAINT fk_tickets_it_employee
        FOREIGN KEY (it_employee_id) REFERENCES employees (id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;
